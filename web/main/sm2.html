SM2 椭圆曲线

GB/T 32918
密钥对 Step 1
生成后，私钥与公钥将自动填入下方所有子功能区。
私钥 hex（64）
公钥 hex（128）

也可手动填入已有私钥 hex，后续所有区块均从上方字段读取；或在各子区单独粘贴。

签名 / 验签 Step 2
用户 ID（留空使用默认 1234567812345678）
待签名消息
随机 k hex（64，留空自动随机）
r hex（64）
s hex（64）
公钥加密 / 解密 GB/T 32918.4

密文格式：C1(64 B 点坐标) + C2(明文长度) + C3(32 B SM3 校验)。点击「填入上方密钥」可快速同步。

分量顺序
加密用公钥 hex（128）
解密用私钥 hex（64）
明文（UTF-8）
加密随机 k（留空自动随机）
密文 hex
ECDH 共享密钥 KDF-SM3

共享点 d·P_peer，对 (x‖y) 做 GB/T KDF 得到 32 字节密钥材料。

本方私钥 hex（64）
对方公钥 hex（128）

提示：私钥与签名仅用于本地体验；生产环境请使用合规密钥设备或 KMS，勿在网页中粘贴真实生产私钥。

在线工具：控件与接口对应

  • 区块：SM2 功能集中在 #sec-sm2（标题「SM2 椭圆曲线」）。
  • 模块就绪：先加载 gm.js，在 Module 可用后声明 cwrap。
  • 密钥：btn-sm2-key → gm_sm2_generate_keypair；输出写入 sm2-priv（64 hex）、sm2-pub（128 hex）。
  • 签名 / 验签：btn-sm2-sign / btn-sm2-verify；随机数 sm2-k；用户标识 sm2-id；结果 sm2-r、sm2-s。
  • 加解密：btn-sm2-enc / btn-sm2-dec；sm2-cipher-order 与库内 order 加解密须一致；密文 sm2-cipher-hex。
  • ECDH：btn-sm2-ecdh → gm_sm2_ecdh_shared_key，输出 sm2-ecdh-out。

export function useSm2Cwrap() {
  const api = useRef(null)
  useEffect(() => {
    let cancelled = false
    gm.default().then((M) => {
      if (cancelled) return
      api.current = {
        M,
        sm2gen: M.cwrap('gm_sm2_generate_keypair', 'number', ['number', 'number', 'number']),
        sm2signFn: M.cwrap('gm_sm2_sign', 'number', ['number', 'number', 'number', 'number', 'number', 'number', 'number', 'number', 'number']),

        sm2ecdhFn: M.cwrap('gm_sm2_ecdh_shared_key', 'number', ['number', 'number', 'number']),
      }
    })
    return () => { cancelled = true }
  }, [])
  return api
}

2. 生成密钥 gm_sm2_generate_keypair

与「生成密钥对」按钮相同：sm2gen 返回非 0 表示成功；私钥 32 字节 → 64 hex，公钥 64 字节（X‖Y）→ 128 hex。

// 熵 32 字节 hex → 作为随机源参与生成
var priv = M._malloc(32), pub = M._malloc(64);
var pe = ac(hexToU8(rndHex(32)));
try {
  if (!sm2gen(pe.ptr, priv, pub)) throw new Error('生成失败');
  var privHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, priv, 32));
  var pubHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, pub, 64));
} finally {
  pe.free();
  M._free(priv);
  M._free(pub);
}
// 假设 Module、sm2gen 已就绪；用 ref 存结果
import { ref } from 'vue'

const privHex = ref(''), pubHex = ref('')

function generateKeypair(M, sm2gen) {
  const priv = M._malloc(32), pub = M._malloc(64)

  try {
    if (!sm2gen(pe.ptr, priv, pub)) throw new Error('生成失败')
    privHex.value = u8ToHex(new Uint8Array(M.HEAPU8.buffer, priv, 32))
    pubHex.value = u8ToHex(new Uint8Array(M.HEAPU8.buffer, pub, 64))
  } finally {
    pe.free()
    M._free(priv)
    M._free(pub)
  }
}

3. 加密 gm_sm2_encrypt

公钥 128 hex、明文 UTF-8、随机 k 64 hex（可空则 rndHex(32)）；order 与下拉框一致，返回值为密文字节数 n。

// C1‖C2‖C3 或 C1‖C3‖C2 由 order 决定，与页面下拉框一致
var pubHex = $('sm2-enc-pub').value.replace(/\s/g, '');
var msg = new TextEncoder().encode($('sm2-plain').value);
var kHex = $('sm2-enc-k').value.replace(/\s/g, '');
if (kHex.length !== 64) {
  kHex = rndHex(32);
  $('sm2-enc-k').value = kHex;
}
var order = parseInt($('sm2-cipher-order').value, 10) || 0;
var cap = 96 + msg.length;
var out = M._malloc(cap);
var ppub = ac(hexToU8(pubHex)), pm = ac(msg), prk = ac(hexToU8(kHex));
try {
  var n = sm2encFn(ppub.ptr, pm.ptr, pm.len, prk.ptr, out, cap, order);
  if (!n) throw new Error('加密失败');
  var cipherHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, out, n));
} finally {
  ppub.free();
  pm.free();
  prk.free();
  M._free(out);
}

5. 签名 gm_sm2_sign

需要私钥/公钥 hex 与随机 k；若填写用户 ID，需把 UTF-8 写入堆并传 idPtr / idLen。

var privHex = $('sm2-priv').value.replace(/\s/g, '');
var pubHex = $('sm2-pub').value.replace(/\s/g, '');
var kHex = $('sm2-k').value.replace(/\s/g, '');
if (kHex.length !== 64) {
  kHex = rndHex(32);
  $('sm2-k').value = kHex;
}
var msg = new TextEncoder().encode($('sm2-msg').value);
var idStr = $('sm2-id').value, idPtr = 0, idLen = 0;
if (idStr) {
  var idU8 = new TextEncoder().encode(idStr);
  idPtr = M._malloc(idU8.length);
  M.HEAPU8.set(idU8, idPtr);
  idLen = idU8.length;
}
var pp = ac(hexToU8(privHex)), ppub = ac(hexToU8(pubHex)), pm = ac(msg), pk = ac(hexToU8(kHex));
var pr = M._malloc(32), ps = M._malloc(32);
try {
  if (!sm2signFn(pp.ptr, pm.ptr, pm.len, ppub.ptr, idPtr, idLen, pk.ptr, pr, ps))
    throw new Error('签名失败');

  var sHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, ps, 32));
} finally {
  pp.free();
  ppub.free();
  pm.free();
  pk.free();
  M._free(pr);
  M._free(ps);

  var sHex = sIn.replace(/\s/g, '')
  var msg = new TextEncoder().encode(msgIn)
  var idStr = idIn, idPtr = 0, idLen = 0
  if (idStr) {
    var idU8 = new TextEncoder().encode(idStr)
    idPtr = M._malloc(idU8.length)
    M.HEAPU8.set(idU8, idPtr)
    idLen = idU8.length
  }
  var ppub = ac(hexToU8(pubHex)), pm = ac(msg), pr2 = ac(hexToU8(rHex)), ps2 = ac(hexToU8(sHex))
  try {
    setVerifyOk(!!sm2verFn(ppub.ptr, pm.ptr, pm.len, idPtr, idLen, pr2.ptr, ps2.ptr))
  } finally {
    ppub.free(); pm.free(); pr2.free(); ps2.free()
    if (idPtr) M._free(idPtr)
  }
}, [api, pubIn, rIn, sIn, msgIn, idIn])
