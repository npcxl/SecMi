SM2 · gm.js

下列示例与页面脚本一致；需 Module、ac、hexToU8 等（见页底）。$('id') 即 document.getElementById('id')。

若页面提示加载失败，请确认项目根目录已生成 gm.js，并尽量通过本地 HTTP 访问（部分浏览器在 file:// 下会限制 WASM）。

在线工具：控件与接口对应

  • 区块：SM2 功能集中在 #sec-sm2（标题「SM2 椭圆曲线」）。
  • 模块就绪：先加载 gm.js，在 Module 可用后声明 cwrap。
  • 密钥：btn-sm2-key → gm_sm2_generate_keypair；输出写入 sm2-priv（64 hex）、sm2-pub（128 hex）。
  • 签名 / 验签：btn-sm2-sign / btn-sm2-verify；随机数 sm2-k；用户标识 sm2-id；结果 sm2-r、sm2-s。
  • 加解密：btn-sm2-enc / btn-sm2-dec；sm2-cipher-order 与库内 order 加解密须一致；密文 sm2-cipher-hex。
  • ECDH：btn-sm2-ecdh → gm_sm2_ecdh_shared_key，输出 sm2-ecdh-out。

1. cwrap 声明

在 Module 就绪后声明；下列为 SM2 相关 7 个导出符号。

var sm2gen = M.cwrap('gm_sm2_generate_keypair','number', ['number','number','number']); var sm2signFn = M.cwrap('gm_sm2_sign', 'number', ['number','number','number','number','number','number','number','number','number']); var sm2verFn = M.cwrap('gm_sm2_verify', 'number', ['number','number','number','number','number','number','number','number']); var sm2encFn = M.cwrap('gm_sm2_encrypt', 'number', ['number','number','number','number','number','number','number']); var sm2decFn = M.cwrap('gm_sm2_decrypt', 'number', ['number','number','number','number','number','number']); var sm2ecdhFn = M.cwrap('gm_sm2_ecdh_shared_key', 'number', ['number','number','number']);

2. 生成密钥 gm_sm2_generate_keypair

与「生成密钥对」按钮相同：sm2gen 返回非 0 表示成功；私钥 32 字节 → 64 hex，公钥 64 字节（X‖Y）→ 128 hex。

var priv = M._malloc(32), pub = M._malloc(64); var pe = ac(hexToU8(rndHex(32))); // 32 字节随机熵 try { if (!sm2gen(pe.ptr, priv, pub)) throw new Error('生成失败'); var privHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, priv, 32)); var pubHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, pub, 64)); } finally { pe.free(); M._free(priv); M._free(pub); }

3. 加密 gm_sm2_encrypt

公钥 128 hex、明文 UTF-8、随机 k 64 hex（可空则自己 rndHex(32)）；order 与下拉框一致，n 为密文字节数。

var pubHex = $('sm2-enc-pub').value.replace(/\s/g, ''); var msg = new TextEncoder().encode($('sm2-plain').value); var kHex = $('sm2-enc-k').value.replace(/\s/g, ''); if (kHex.length !== 64) { kHex = rndHex(32); $('sm2-enc-k').value = kHex; } var order = parseInt($('sm2-cipher-order').value, 10) || 0; var cap = 96 + msg.length; var out = M._malloc(cap); var ppub = ac(hexToU8(pubHex)), pm = ac(msg), prk = ac(hexToU8(kHex)); try { var n = sm2encFn(ppub.ptr, pm.ptr, pm.len, prk.ptr, out, cap, order); if (!n) throw new Error('加密失败'); var cipherHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, out, n)); } finally { ppub.free(); pm.free(); prk.free(); M._free(out); }

4. 解密 gm_sm2_decrypt

密文总长至少 96 字节；order 必须与加密时相同。

var privHex = $('sm2-dec-priv').value.replace(/\s/g, ''); var hex = $('sm2-cipher-hex').value.replace(/\s/g, ''); var cipher = hexToU8(hex); if (cipher.length < 96) throw new Error('密文太短'); var mlen = cipher.length - 96; var order = parseInt($('sm2-cipher-order').value, 10) || 0; var out = M._malloc(Math.max(mlen, 1)); var ppriv = ac(hexToU8(privHex)), pc = ac(cipher); try { var n = sm2decFn(ppriv.ptr, pc.ptr, pc.len, out, mlen, order); if (!n) throw new Error('解密失败'); var plain = new TextDecoder().decode(new Uint8Array(M.HEAPU8.buffer, out, n)); } finally { ppriv.free(); pc.free(); M._free(out); }

5. 签名 gm_sm2_sign

需要完整私钥/公钥 hex 与随机 k；若填写了用户 ID，需把字符串拷进堆并传 idPtr / idLen。

var privHex = $('sm2-priv').value.replace(/\s/g, ''); var pubHex = $('sm2-pub').value.replace(/\s/g, ''); var kHex = $('sm2-k').value.replace(/\s/g, ''); if (kHex.length !== 64) { kHex = rndHex(32); $('sm2-k').value = kHex; } var msg = new TextEncoder().encode($('sm2-msg').value); var idStr = $('sm2-id').value, idPtr = 0, idLen = 0; if (idStr) { var idU8 = new TextEncoder().encode(idStr); idPtr = M._malloc(idU8.length); M.HEAPU8.set(idU8, idPtr); idLen = idU8.length; } var pp = ac(hexToU8(privHex)), ppub = ac(hexToU8(pubHex)), pm = ac(msg), pk = ac(hexToU8(kHex)); var pr = M._malloc(32), ps = M._malloc(32); try { if (!sm2signFn(pp.ptr, pm.ptr, pm.len, ppub.ptr, idPtr, idLen, pk.ptr, pr, ps)) throw new Error('签名失败'); var rHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, pr, 32)); var sHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, ps, 32)); } finally { pp.free(); ppub.free(); pm.free(); pk.free(); M._free(pr); M._free(ps); if (idPtr) M._free(idPtr); }

6. 验签 gm_sm2_verify

返回值为真/假；用户 ID 处理与签名相同。

var pubHex = $('sm2-pub').value.replace(/\s/g, ''); var rHex = $('sm2-r').value.replace(/\s/g, ''); var sHex = $('sm2-s').value.replace(/\s/g, ''); var msg = new TextEncoder().encode($('sm2-msg').value); var idStr = $('sm2-id').value, idPtr = 0, idLen = 0; if (idStr) { var idU8 = new TextEncoder().encode(idStr); idPtr = M._malloc(idU8.length); M.HEAPU8.set(idU8, idPtr); idLen = idU8.length; } var ppub = ac(hexToU8(pubHex)), pm = ac(msg), pr2 = ac(hexToU8(rHex)), ps2 = ac(hexToU8(sHex)); try { var ok = sm2verFn(ppub.ptr, pm.ptr, pm.len, idPtr, idLen, pr2.ptr, ps2.ptr); } finally { ppub.free(); pm.free(); pr2.free(); ps2.free(); if (idPtr) M._free(idPtr); }

7. ECDH gm_sm2_ecdh_shared_key

本方私钥 64 hex、对方公钥 128 hex；输出 32 字节共享密钥 material（hex）。

var privHex = $('sm2-ecdh-priv').value.replace(/\s/g, ''); var peerHex = $('sm2-ecdh-peer').value.replace(/\s/g, ''); var out = M._malloc(32); var ppriv = ac(hexToU8(privHex)), ppeer = ac(hexToU8(peerHex)); try { if (!sm2ecdhFn(ppriv.ptr, ppeer.ptr, out)) throw new Error('ECDH 失败'); var keyHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, out, 32)); } finally { ppriv.free(); ppeer.free(); M._free(out); }
