在线工具：控件与接口对应

  • 区块：SM4 功能在 #sec-sm4（标题「SM4 分组密码」）。
  • S 盒：sm4-sbox → 加解密前调用 gm_sm4_set_sbox_version（0 现行，1 早期印刷版）。
  • CBC 加密：btn-sm4-enc → gm_sm4_cbc_encrypt；密钥 sm4-key、IV sm4-iv、明文 sm4-plain；密文 hex 写入 sm4-cipher-hex。
  • CBC 解密：btn-sm4-dec → gm_sm4_cbc_decrypt；密文来自 sm4-cipher-hex。

1. cwrap 声明

在 Module 就绪后声明；下列为 SM4 CBC 与 S 盒相关导出。

// S 盒版本 + CBC PKCS#7（与页底脚本一致）
var sm4SetSbox = M.cwrap('gm_sm4_set_sbox_version', null, ['number']);
var sm4CbcEnc = M.cwrap('gm_sm4_cbc_encrypt', 'number', ['number', 'number', 'number', 'number', 'number']);
var sm4CbcDec = M.cwrap('gm_sm4_cbc_decrypt', 'number', ['number', 'number', 'number', 'number', 'number', 'number']);
import { shallowRef, onMounted } from 'vue'
import * as gm from './gm.js'

const Module = shallowRef(null)
const sm4SetSbox = shallowRef(null)
const sm4CbcEnc = shallowRef(null)
const sm4CbcDec = shallowRef(null)

onMounted(async () => {
  const M = await gm.default()
  Module.value = M
  sm4SetSbox.value = M.cwrap('gm_sm4_set_sbox_version', null, ['number'])
  sm4CbcEnc.value = M.cwrap('gm_sm4_cbc_encrypt', 'number', ['number', 'number', 'number', 'number', 'number'])
  sm4CbcDec.value = M.cwrap('gm_sm4_cbc_decrypt', 'number', ['number', 'number', 'number', 'number', 'number', 'number'])
})
import { useEffect, useRef } from 'react'
import * as gm from './gm.js'

export function useSm4Cwrap() {
  const api = useRef(null)
  useEffect(() => {
    let cancelled = false
    gm.default().then((M) => {
      if (cancelled) return
      api.current = {
        M,
        sm4SetSbox: M.cwrap('gm_sm4_set_sbox_version', null, ['number']),
        sm4CbcEnc: M.cwrap('gm_sm4_cbc_encrypt', 'number', ['number', 'number', 'number', 'number', 'number']),
        sm4CbcDec: M.cwrap('gm_sm4_cbc_decrypt', 'number', ['number', 'number', 'number', 'number', 'number', 'number']),
      }
    })
    return () => { cancelled = true }
  }, [])
  return api
}

2. CBC 加密 gm_sm4_cbc_encrypt

密钥与 IV 各 16 字节（32 hex）；明文为 UTF-8；返回值为密文字节长度。

sm4SetSbox(parseInt($('sm4-sbox').value, 10) || 0);
var key = hexToU8($('sm4-key').value.replace(/\s/g, ''));
var iv = hexToU8($('sm4-iv').value.replace(/\s/g, ''));
var plain = new TextEncoder().encode($('sm4-plain').value);
var pin = ac(plain), pk = ac(key), piv = ac(iv);
var outCap = plain.length + 32;
var outPtr = M._malloc(outCap);
try {
  var n = sm4CbcEnc(pk.ptr, piv.ptr, pin.ptr, pin.len, outPtr);
  if (!n) throw new Error('加密失败');
  var cipherHex = u8ToHex(new Uint8Array(M.HEAPU8.buffer, outPtr, n));
} finally {
  pin.free(); pk.free(); piv.free(); M._free(outPtr);
}
import { ref } from 'vue'

const sboxVer = ref(0), keyHex = ref(''), ivHex = ref(''), plainIn = ref('')
const cipherHex = ref('')

function cbcEncrypt(M, sm4SetSbox, sm4CbcEnc) {
  sm4SetSbox(sboxVer.value || 0)
  var key = hexToU8(keyHex.value.replace(/\s/g, ''))
  var iv = hexToU8(ivHex.value.replace(/\s/g, ''))
  var plain = new TextEncoder().encode(plainIn.value)
  var pin = ac(plain), pk = ac(key), piv = ac(iv)
  var outCap = plain.length + 32
  var outPtr = M._malloc(outCap)
  try {
    var n = sm4CbcEnc(pk.ptr, piv.ptr, pin.ptr, pin.len, outPtr)
    if (!n) throw new Error('加密失败')
    cipherHex.value = u8ToHex(new Uint8Array(M.HEAPU8.buffer, outPtr, n))
  } finally {
    pin.free(); pk.free(); piv.free(); M._free(outPtr)
  }
}
import { useCallback, useState } from 'react'

const api = useSm4Cwrap()
const [sboxVer, setSboxVer] = useState(0)
const [keyHex, setKeyHex] = useState('')
const [ivHex, setIvHex] = useState('')
const [plainIn, setPlainIn] = useState('')
const [cipherHex, setCipherHex] = useState('')

const cbcEncrypt = useCallback(() => {
  const { M, sm4SetSbox, sm4CbcEnc } = api.current
  if (!M || !sm4SetSbox || !sm4CbcEnc) return
  sm4SetSbox(sboxVer || 0)
  var key = hexToU8(keyHex.replace(/\s/g, ''))
  var iv = hexToU8(ivHex.replace(/\s/g, ''))
  var plain = new TextEncoder().encode(plainIn)
  var pin = ac(plain), pk = ac(key), piv = ac(iv)
  var outCap = plain.length + 32
  var outPtr = M._malloc(outCap)
  try {
    var n = sm4CbcEnc(pk.ptr, piv.ptr, pin.ptr, pin.len, outPtr)
    if (!n) throw new Error('加密失败')
    setCipherHex(u8ToHex(new Uint8Array(M.HEAPU8.buffer, outPtr, n)))
  } finally {
    pin.free(); pk.free(); piv.free(); M._free(outPtr)
  }
}, [api, sboxVer, keyHex, ivHex, plainIn])

3. CBC 解密 gm_sm4_cbc_decrypt

密文为 hex；输出长度不超过 out_cap，成功时返回明文字节数。

sm4SetSbox(parseInt($('sm4-sbox').value, 10) || 0);
var cipher = hexToU8($('sm4-cipher-hex').value.replace(/\s/g, ''));
var pk = ac(hexToU8($('sm4-key').value.replace(/\s/g, '')));
var piv = ac(hexToU8($('sm4-iv').value.replace(/\s/g, '')));
var pc = ac(cipher);
var outCap = cipher.length;
var outPtr = M._malloc(outCap);
try {
  var n = sm4CbcDec(pk.ptr, piv.ptr, pc.ptr, pc.len, outPtr, outCap);
  if (!n) throw new Error('解密失败');
  var plainStr = new TextDecoder().decode(new Uint8Array(M.HEAPU8.buffer, outPtr, n));
} finally {
  pk.free(); piv.free(); pc.free(); M._free(outPtr);
}
import { ref } from 'vue'

const sboxVer = ref(0), keyHex = ref(''), ivHex = ref('')
const cipherHexIn = ref('')
const plainOut = ref('')

function cbcDecrypt(M, sm4SetSbox, sm4CbcDec) {
  sm4SetSbox(sboxVer.value || 0)
  var cipher = hexToU8(cipherHexIn.value.replace(/\s/g, ''))
  var pk = ac(hexToU8(keyHex.value.replace(/\s/g, '')))
  var piv = ac(hexToU8(ivHex.value.replace(/\s/g, '')))
  var pc = ac(cipher)
  var outCap = cipher.length
  var outPtr = M._malloc(outCap)
  try {
    var n = sm4CbcDec(pk.ptr, piv.ptr, pc.ptr, pc.len, outPtr, outCap)
    if (!n) throw new Error('解密失败')
    plainOut.value = new TextDecoder().decode(new Uint8Array(M.HEAPU8.buffer, outPtr, n))
  } finally {
    pk.free(); piv.free(); pc.free(); M._free(outPtr)
  }
}
import { useCallback, useState } from 'react'

const api = useSm4Cwrap()
const [sboxVer, setSboxVer] = useState(0)
const [keyHex, setKeyHex] = useState('')
const [ivHex, setIvHex] = useState('')
const [cipherHexIn, setCipherHexIn] = useState('')
const [plainOut, setPlainOut] = useState('')

const cbcDecrypt = useCallback(() => {
  const { M, sm4SetSbox, sm4CbcDec } = api.current
  if (!M || !sm4SetSbox || !sm4CbcDec) return
  sm4SetSbox(sboxVer || 0)
  var cipher = hexToU8(cipherHexIn.replace(/\s/g, ''))
  var pk = ac(hexToU8(keyHex.replace(/\s/g, '')))
  var piv = ac(hexToU8(ivHex.replace(/\s/g, '')))
  var pc = ac(cipher)
  var outCap = cipher.length
  var outPtr = M._malloc(outCap)
  try {
    var n = sm4CbcDec(pk.ptr, piv.ptr, pc.ptr, pc.len, outPtr, outCap)
    if (!n) throw new Error('解密失败')
    setPlainOut(new TextDecoder().decode(new Uint8Array(M.HEAPU8.buffer, outPtr, n)))
  } finally {
    pk.free(); piv.free(); pc.free(); M._free(outPtr)
  }
}, [api, sboxVer, keyHex, ivHex, cipherHexIn])
